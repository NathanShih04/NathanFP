import java.util.Random;
int flipCount = 10;

public void coinFlip(int count){
    Random rando = new Random();
    for(int i = 0; i < count; i++){
        int side = rando.nextInt(2);
        if(side == 1){
            System.out.println("Heads");
        }
        else{
            System.out.println("Tails");
        }
    }
}

coinFlip(flipCount);
